C#
Now let's create a MessageClient.cs that we will use to communicate with our WebApi project
keep in mind that dapr-demo-api is the name on the tye.yaml or the —app-id name pass to dapr cli
public class MessageClient { private readonly string uriServiceInvocation = $"/v1.0/invoke/dapr-demo-api/method"; private readonly HttpClient client; public MessageClient(HttpClient client) { this.client = client; } public async Task<String> GeMessageAsyc(string id) { var responseMessage = await this.client.GetAsync($"{uriServiceInvocation}/api/message?id={id}"); responseMessage.EnsureSuccessStatusCode(); var data = await responseMessage.Content.ReadAsStringAsync(); return data; } }
C#
And WeatherClient.cs
public class WeatherClient { private readonly JsonSerializerOptions options = new JsonSerializerOptions() { PropertyNamingPolicy = JsonNamingPolicy.CamelCase, PropertyNameCaseInsensitive = true, }; private readonly HttpClient client; public WeatherClient(HttpClient client) { this.client = client; } public async Task<WeatherForecast[]> GetWeatherAsyc() { var responseMessage = await this.client.GetAsync("/v1.0/invoke/dapr-demo-api/method/weatherforecast"); var stream = await responseMessage.Content.ReadAsStreamAsync(); return await JsonSerializer.DeserializeAsync<WeatherForecast[]>(stream, options); } }
C#
Now Let's add AddHttpClient() to the ConfigureServices
public void ConfigureServices(IServiceCollection services) { services.AddRazorPages(); services.AddHttpClient<MessageClient>(client => { client.BaseAddress = new Uri($"http://localhost:{Environment.GetEnvironmentVariable("DAPR_HTTP_PORT")}"); }); services.AddHttpClient<WeatherClient>(client => { client.BaseAddress = new Uri($"http://localhost:{Environment.GetEnvironmentVariable("DAPR_HTTP_PORT")}"); }); }
C#
Now Let call our clients from the razor page behind Index.cshtml.cs
public class IndexModel : PageModel { public WeatherForecast[] Forecasts { get; set; } public String Message { get; set; } private readonly ILogger<IndexModel> _logger; public IndexModel(ILogger<IndexModel> logger) { _logger = logger; } public async Task OnGet([FromServices] MessageClient client, [FromServices] WeatherClient wClient) { Message = await client.GeMessageAsyc("test2"); Forecasts = await wClient.GetWeatherAsyc(); } }
